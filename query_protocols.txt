SELECT rc.* 
FROM [CursorImport].[import].[RegionalCommon] rc 
join [CursorImport].[import].Regional_Notif rn on rc.Local_Notif_ID=rn.NotifNr 
join [Cursor].dbo.Tender t on rn.NotifNr=t.NotifNr 
join [Cursor].dbo.Lot l on t.Tender_ID=l.Tender_ID 
where TypeReasonFail_ID is null and Winner_ID is null and t.PublDt > cast(dateadd(MM, -4, getdate()) as date)
